  show_grid: true;
});

  // 2d array of x,y values ex:  [[1,1], [2,3], [3,20]]

  data: create_squared_points() 
})

  data: create_exponential_points() 
                                   
  options: {
    line_color : "#55bb00"
    area_color : "#55bb00"
    dot_color  :  "#55bb00"
  }
}

c.draw()

Label Formatting

  • %m - month (1-12)
  • %d - date (1-31)
  • %Y - year (eg. 2011)
  • %H - hours (0-24)
  • %I - hours (1-12)
  • %p - am/pm indicator
  • %M - minutes (0-59)

Example: label_format: "%m/%d/%Y" -> "11/05/2011"

Lots of Data

  show_grid: true,
  label_format: "%I:%M %p",
  x_label_size: 11,
  show_y_labels: true,
  y_label_size: 11,
  smoothing: 1,
  max_x_labels: 5,
  max_y_labels: 10,
  x_padding: 50,
})
