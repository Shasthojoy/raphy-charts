Easy, beautiful line charts with sane defaults

Installation

  1. Make sure you have Raphael.js loaded on your page
  2. Download Raphy Charts and place it on your page

Basic Usage


// 'chart1' is the id of an existing dom element
// with a defined width/height. DOM is assumed to be ready 
c = new Charts.LineChart('chart1')

c.add_line {
  data: create_exponential_points() # array of numbers
  labels: create_labels()           # array of strings or date objects 
  options: {
    line_color : "#cc1100",
    area_color : "#cc1100",
    dot_color  :  "#cc1100"
  }
}
c.add_line { data: create_squared_points() }
c.draw()

Smoothed Line Chart


c = new Charts.LineChart('chart2', {
  line_color : "#118800",
  dot_color  : "#118800",
  area_color : "#118800",
  dot_stroke_color: "#aaa",
  dot_stroke_size: 3, 
  label_min  : false,
  smoothing  : 0.5,
  show_grid  : true
})

c.add_line({
  data: create_random_points2()
})

c.draw()
