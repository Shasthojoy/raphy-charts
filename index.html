Raphy Charts Download Raphy Charts

Easy, beautiful line charts with sane defaults

Installation

  1. Make sure you have Raphael.js loaded on your page
  2. Download Raphy Charts and place it on your page

Basic Usage


// mydiv1 is the id of an existing dom element
// with a defined width/height 
var chart = new Charts.LineChart("mydiv1")

chart.add_line({
  data: [1,2,3,4,5],
  labels: ["mon", "tues", "wed", "thurs", "fri"],
  options: {
    // Extra Options...see below
  }
})

chart.draw()

Area Chart


c = new Charts.LineChart('chart1')

c.add_line {
  data: create_exponential_points() # array of numbers
  labels: create_labels()           # array of date objects 
  options: {
    line_color : "#cc1100",
    area_color : "#cc1100",
    dot_color  :  "#cc1100"
  }
}
c.add_line { data: create_squared_points() }
c.draw()

Smoothed Line Chart


c = new Charts.LineChart('chart2', {
  line_color : "#118800",
  dot_color  : "#118800",
  area_color : "#118800",
  dot_stroke_color: "#aaa",
  dot_stroke_size: 3, 
  label_min  : false,
  smoothing  : 0.5,
  show_grid  : true
})

c.add_line({
  data: create_random_points2()
})

c.draw()
