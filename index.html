Raphy Charts Download

Easy, beautiful line charts with sane defaults

Installation

  1. Make sure you have Raphael.js loaded on your page
  2. Download Raphy Charts and place it on your page

Usage


/* 
 * DOM is assumed to be ready 
 * 'chart1' is the id of an existing DOM element
 *  with a defined width/height. Eg. div id="chart1" style="width:100px; height:100px;"
*/
c = new Charts.LineChart('chart1', {
  show_grid: true
})

c.add_line {
  data: create_exponential_points() // 2d array of x,y values ex:  [[1,1], [2,3], [3,20]]
                                   // OR x values may be dates: [[new Date(),100]...]
  options: {
    line_color : "#cc1100",
    area_color : "#cc1100",
    dot_color  :  "#cc1100"
  }
}
c.add_line { data: create_squared_points() }
c.draw()

Label Formatting

  • %m - month (1-12)
  • %d - date (1-31)
  • %Y - year (eg. 2011)
  • %H - hours (0-24)
  • %I - hours (1-12 am/pm)
  • %M - minutes (0-59)

Example: label_format: "%m/%d/%Y" -> "11/05/2011"

Smoothed Line Chart


c = new Charts.LineChart('chart2', {
  line_color : "#118800",
  dot_color  : "#118800",
  area_color : "#118800",
  dot_stroke_color: "#aaa",
  dot_stroke_size: 3, 
  label_min  : false,
  smoothing  : 0.5,
  show_grid  : true
})

c.add_line({
  data: create_random_points2()
})

c.draw()
